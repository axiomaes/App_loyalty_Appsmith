-- Cambia la contraseña. No toca rol ni otros campos.
-- Aísla por userId + businessId para seguridad multi-tenant.

UPDATE "User"
SET password = crypt({{ this.params.newPassword }}, gen_salt('bf', 12))
WHERE id = {{ appsmith.store.userId }}::uuid
  AND "businessId" = {{ Auth.businessId() }}::uuid
RETURNING id, email, role;
